    hid = input("–í–∞—à —Ö—ñ–¥: ").lower()

def computer_hodyt() -> tuple[int, str]:
      return choice([(0, "–ö–∞–º—ñ–Ω—åü™®"),
                     (1, "–ù–æ–∂–∏—Ü—ñ ‚úÇ"),
                     (2, "–ü–∞–ø—ñ—Äüìú")])

def ocinka_hodu(n1: int, n2: int) -> tuple[str, str]:
    match n1, n2:
        case (0, 1) | (1, 2) | (2, 0):

        case _:
